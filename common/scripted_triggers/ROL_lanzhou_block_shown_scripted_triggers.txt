ROL_KMT_lanzhou_block_1_visible_trigger = {
	OR = {
		has_country_flag = ROL_KMT_1_button_use
		has_political_power < 0
		AND = {
			check_variable = {
                ROL_KMT_lanzhou_stage = 1
            }
			OR = {
				has_political_power < 10.01
				check_variable = { ROL_KMT_lanzhou_manpower_num < 3.01 }
				check_variable = { ROL_KMT_lanzhou_suppor_num < 3.01 }
			}
		}
		AND = {
			check_variable = {
                ROL_KMT_lanzhou_stage = 2
            }
			OR = {
				has_political_power < 20.01
				check_variable = { ROL_KMT_lanzhou_manpower_num < 5.01 }
				NOT = {
					has_completed_focus = ROL_KMT_responsibility_distinguish
				}
			}
		}
		AND = {
			check_variable = {
                ROL_KMT_lanzhou_stage = 3
            }
			OR = {
				has_political_power < 20.01
				check_variable = { ROL_KMT_lanzhou_suppor_num < 20.01 }
				has_equipment = { infantry_equipment < 500 }
				NOT = {
					has_completed_focus = ROL_KMT_supply_sufficient
				}
			}
		}
	}
}
