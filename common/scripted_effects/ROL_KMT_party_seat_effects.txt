ROL_KMT_operate_party_seat = {
	#改变席位
    add_to_variable = {
        var = ROL_KMT_operate_party_seat_points #变量
    }
    
    if = {
        limit = {
            check_variable = {
                ROL_KMT_operate_party_seat_points_temp < 0
            }
        }
        multiply_temp_variable = {
            ROL_KMT_operate_party_seat_points_temp = -1
        }
    }
    else = {
    }
    ######
    clear_variable = ROL_KMT_operate_party_seat_points_temp
    ######
    clamp_variable = {
		var = ROL_KMT_add_deduct_points
		min = 1
		max = 25
	}
}

ROL_KMT_CBI_Leveler = {
	if = {
		limit = {
			check_variable = {
				OR = {
					ROL_KMT_party_CBI_green_num > 25
					ROL_KMT_party_CBI_green_num = 25
				}	
				ROL_KMT_party_CBI_green_num < 50
			}
		}
		swap_ideas = {
			add_idea = KMT_ROL_CBI_1_idea
		}
	}
	else_if = {
		limit = {
			check_variable = {
				OR = {
					ROL_KMT_party_CBI_green_num > 50
					ROL_KMT_party_CBI_green_num = 50
				}	
				ROL_KMT_party_CBI_green_num < 75
			}
		}
		swap_ideas = {
			remove_idea = KMT_ROL_CBI_1_idea
			add_idea = KMT_ROL_CBI_2_idea
		}
	}
	else_if = {
		limit = {
			check_variable = {
				OR = {
					ROL_KMT_party_CBI_green_num > 75
					ROL_KMT_party_CBI_green_num = 75
				}	
				ROL_KMT_party_CBI_green_num < 100
			}
		}
		swap_ideas = {
			remove_idea = KMT_ROL_CBI_2_idea
			add_idea = KMT_ROL_CBI_3_idea
		}
	}
	else_if = {
		limit = {
			check_variable = {
				ROL_KMT_party_CBI_green_num = 100
			}
		}
		swap_ideas = {
			remove_idea = KMT_ROL_CBI_3_idea
			add_idea = KMT_ROL_CBI_4_idea
		}
	}
}
